import { ref, onMounted } from 'vue'

// FAKE SITE
const site = {
  name: "Kasarani Micro-Grid",
  location: "Nairobi County"
}

// DYNAMIC STATS
const generation = ref(4.2)
const load = ref(3.1)

const irradiance = ref(800) // W/m²
const output = ref(3.5)
const battery = ref(70)

onMounted(() => {
  setInterval(() => {
    // simulate sun fluctuation
    irradiance.value = (600 + Math.random() * 400).toFixed(0)

    // simple physics-ish relation
    output.value = (irradiance.value * 0.004).toFixed(2)

    // battery reacts slowly
    battery.value = Math.max(20, Math.min(100,
      battery.value + (output.value > 2 ? 0.5 : -0.3)
    )).toFixed(0)

  }, 2000)
})





// LOGS
const logs = ref([
  "System initialized",
  "Battery stable",
  "Load within limits"
])

// FAKE LIVE UPDATES
onMounted(() => {
  setInterval(() => {
    generation.value = (3 + Math.random() * 2).toFixed(1)
    load.value = (2 + Math.random() * 2).toFixed(1)
    battery.value = Math.max(20, Math.min(100, battery.value + (Math.random() - 0.5) * 2)).toFixed(0)

    logs.value.unshift("Load adjusted at " + new Date().toLocaleTimeString())
    if (logs.value.length > 5) logs.value.pop()
  }, 2000)
})
</script>

<script setup>
import { onMounted } from 'vue'

onMounted(() => {
  const canvas = document.getElementById("sphereCanvas")
  if (!canvas) return

  const ctx = canvas.getContext("2d")
  const size = 350 // Match the SCSS size
  canvas.width = size
  canvas.height = size
  const center = size / 2

  // --- Core Sphere Config ---
  const radius = 100
  const totalRings = 10
  const stepsPerRing = 18

  // --- Dynamic Points (The 'Fancy' part) ---
  const dynamicPoints = []
  const maxDynamicPoints = 15 // Keep it sparse

  // Pre-generate static vertices
  const vertices = []
  for (let i = 0; i <= totalRings; i++) {
    const lat = Math.PI * (i / totalRings - 0.5) // Latitude -90 to 90
    for (let j = 0; j < stepsPerRing; j++) {
      const lon = 2 * Math.PI * (j / stepsPerRing) // Longitude 0 to 360
      const x = radius * Math.cos(lat) * Math.cos(lon)
      const y = radius * Math.cos(lat) * Math.sin(lon)
      const z = radius * Math.sin(lat)
      vertices.push({ x, y, z })
    }
  }

  // Generate real random 3D points *on* the sphere surface
  function generateDynamicPoint() {
    const phi = Math.random() * Math.PI * 2;
    const theta = Math.acos((Math.random() * 2) - 1);
    const pointRadius = radius; // Start on surface

    return {
      x: pointRadius * Math.sin(theta) * Math.cos(phi),
      y: pointRadius * Math.sin(theta) * Math.sin(phi),
      z: pointRadius * Math.cos(theta),
      life: Math.random() * 0.5 + 0.5, // 0.5s to 1s life
      pulseFactor: Math.random() * 2 // How much it 'projects'
    }
  }

  let angle = 0

  function draw() {
    ctx.clearRect(0, 0, canvas.width, canvas.height)
    angle += 0.005 // Slower, purposeful spin

    // --- 1. Project and Rotate all Static Vertices ---
    const projected2D = vertices.map(v => {
      // Basic 3D rotation matrix application
      // Rotate Y (Vertical)
      const ryX = v.x * Math.cos(angle) + v.z * Math.sin(angle)
      const ryZ = -v.x * Math.sin(angle) + v.z * Math.cos(angle)

      // Rotate X (Horizontal)
      const yX = ryZ * Math.sin(angle) + v.y * Math.cos(angle)
      const yZ = ryZ * Math.cos(angle) - v.y * Math.sin(angle)

      // Perspective projection (simple, good enough for wireframe)
      const scale = 200 / (200 + yZ)
      const screenX = ryX * scale + center
      const screenY = yX * scale + center
      return { x: screenX, y: screenY, z: yZ }
    })

    // --- 2. Draw Wireframe Lines ---
    ctx.strokeStyle = "rgba(255, 255, 255, 0.15)"
    ctx.lineWidth = 0.5
    ctx.beginPath()

    // Draw Latitudes
    for (let i = 0; i < totalRings; i++) {
      const start = i * stepsPerRing
      ctx.moveTo(projected2D[start].x, projected2D[start].y)
      for (let j = 1; j < stepsPerRing; j++) {
        ctx.lineTo(projected2D[start + j].x, projected2D[start + j].y)
      }
      ctx.lineTo(projected2D[start].x, projected2D[start].y) // Close ring
    }

    // Draw Longitudes
    for (let j = 0; j < stepsPerRing; j++) {
      ctx.moveTo(projected2D[j].x, projected2D[j].y)
      for (let i = 1; i < totalRings; i++) {
        ctx.lineTo(projected2D[i * stepsPerRing + j].x, projected2D[i * stepsPerRing + j].y)
      }
    }
    ctx.stroke()

    // --- 3. Manage and Draw Dynamic 'Micro-grid' Points ---
    if (dynamicPoints.length < maxDynamicPoints && Math.random() > 0.95) {
      dynamicPoints.push(generateDynamicPoint());
    }

    for (let i = dynamicPoints.length - 1; i >= 0; i--) {
      const p = dynamicPoints[i];
      p.life -= 0.01; // Fade life

      if (p.life <= 0) {
        dynamicPoints.splice(i, 1);
        continue;
      }

      // Project the dynamic point (with the current rotation)
      const ryX = p.x * Math.cos(angle) + p.z * Math.sin(angle)
      const ryZ = -p.x * Math.sin(angle) + p.z * Math.cos(angle)
      const yX = ryZ * Math.sin(angle) + p.y * Math.cos(angle)
      const yZ = ryZ * Math.cos(angle) - p.y * Math.cos(angle)

      const scale = 200 / (200 + yZ)
      const screenX = ryX * scale + center
      const screenY = yX * scale + center

      // Calculate opacity based on life and depth (Z)
      const depthAlpha = Math.max(0, (yZ + radius) / (radius * 2)); // Faint on back
      const opacity = p.life * depthAlpha * 0.7;

      if (opacity <= 0.05) continue; // Skip hidden/faded points

      // Draw the point (a faint blue pulse)
      ctx.fillStyle = `rgba(0, 255, 255, ${opacity})`;
      ctx.beginPath();
      // Use the pulseFactor to subtly vary the point size
      ctx.arc(screenX, screenY, 1.5 + (1 - p.life) * p.pulseFactor, 0, Math.PI * 2);
      ctx.fill();

      // Fancy connection: Find *one* nearest neighbor vertex and connect with a faint line
      let minDst = 1000;
      let targetV = null;
      for(let j=0; j<projected2D.length; j++) {
        const v = projected2D[j];
        if (v.z > p.z) continue; // Only connect 'down' in depth
        const dst = Math.pow(v.x-screenX, 2) + Math.pow(v.y-screenY, 2);
        if(dst < minDst && dst < 500) { // arbitrary limit to prevent long crazy lines
          minDst = dst;
          targetV = v;
        }
      }

      if(targetV) {
        ctx.strokeStyle = `rgba(0, 255, 255, ${opacity * 0.9})`;
        ctx.lineWidth = 0.5;
        ctx.beginPath();
        ctx.moveTo(screenX, screenY);
        ctx.lineTo(targetV.x, targetV.y);
        ctx.stroke();
      }
    }

    requestAnimationFrame(draw)
  }

  draw()
})
</script>

<script setup>
import { onMounted } from 'vue'

onMounted(() => {
  const canvas = document.getElementById("sphereCanvas")
  if (!canvas) return

  const ctx = canvas.getContext("2d")
  const size = 350 // Match the SCSS size
  canvas.width = size
  canvas.height = size
  const center = size / 2

  // --- Core Sphere Config ---
  const radius = 100
  const totalRings = 10
  const stepsPerRing = 18

  // --- Dynamic Points (The 'Fancy' part) ---
  const dynamicPoints = []
  const maxDynamicPoints = 15 // Keep it sparse

  // Pre-generate static vertices
  const vertices = []
  for (let i = 0; i <= totalRings; i++) {
    const lat = Math.PI * (i / totalRings - 0.5) // Latitude -90 to 90
    for (let j = 0; j < stepsPerRing; j++) {
      const lon = 2 * Math.PI * (j / stepsPerRing) // Longitude 0 to 360
      const x = radius * Math.cos(lat) * Math.cos(lon)
      const y = radius * Math.cos(lat) * Math.sin(lon)
      const z = radius * Math.sin(lat)
      vertices.push({ x, y, z })
    }
  }

  // Generate real random 3D points *on* the sphere surface
  function generateDynamicPoint() {
    const phi = Math.random() * Math.PI * 2;
    const theta = Math.acos((Math.random() * 2) - 1);
    const pointRadius = radius; // Start on surface

    return {
      x: pointRadius * Math.sin(theta) * Math.cos(phi),
      y: pointRadius * Math.sin(theta) * Math.sin(phi),
      z: pointRadius * Math.cos(theta),
      life: Math.random() * 0.5 + 0.5, // 0.5s to 1s life
      pulseFactor: Math.random() * 2 // How much it 'projects'
    }
  }

  let angle = 0

  function draw() {
    ctx.clearRect(0, 0, canvas.width, canvas.height)
    angle += 0.005 // Slower, purposeful spin

    // --- 1. Project and Rotate all Static Vertices ---
    const projected2D = vertices.map(v => {
      // Basic 3D rotation matrix application
      // Rotate Y (Vertical)
      const ryX = v.x * Math.cos(angle) + v.z * Math.sin(angle)
      const ryZ = -v.x * Math.sin(angle) + v.z * Math.cos(angle)

      // Rotate X (Horizontal)
      const yX = ryZ * Math.sin(angle) + v.y * Math.cos(angle)
      const yZ = ryZ * Math.cos(angle) - v.y * Math.sin(angle)

      // Perspective projection (simple, good enough for wireframe)
      const scale = 200 / (200 + yZ)
      const screenX = ryX * scale + center
      const screenY = yX * scale + center
      return { x: screenX, y: screenY, z: yZ }
    })

    // --- 2. Draw Wireframe Lines ---
    ctx.strokeStyle = "rgba(255, 255, 255, 0.15)"
    ctx.lineWidth = 0.5
    ctx.beginPath()

    // Draw Latitudes
    for (let i = 0; i < totalRings; i++) {
      const start = i * stepsPerRing
      ctx.moveTo(projected2D[start].x, projected2D[start].y)
      for (let j = 1; j < stepsPerRing; j++) {
        ctx.lineTo(projected2D[start + j].x, projected2D[start + j].y)
      }
      ctx.lineTo(projected2D[start].x, projected2D[start].y) // Close ring
    }

    // Draw Longitudes
    for (let j = 0; j < stepsPerRing; j++) {
      ctx.moveTo(projected2D[j].x, projected2D[j].y)
      for (let i = 1; i < totalRings; i++) {
        ctx.lineTo(projected2D[i * stepsPerRing + j].x, projected2D[i * stepsPerRing + j].y)
      }
    }
    ctx.stroke()

    // --- 3. Manage and Draw Dynamic 'Micro-grid' Points ---
    if (dynamicPoints.length < maxDynamicPoints && Math.random() > 0.95) {
      dynamicPoints.push(generateDynamicPoint());
    }

    for (let i = dynamicPoints.length - 1; i >= 0; i--) {
      const p = dynamicPoints[i];
      p.life -= 0.01; // Fade life

      if (p.life <= 0) {
        dynamicPoints.splice(i, 1);
        continue;
      }

      // Project the dynamic point (with the current rotation)
      const ryX = p.x * Math.cos(angle) + p.z * Math.sin(angle)
      const ryZ = -p.x * Math.sin(angle) + p.z * Math.cos(angle)
      const yX = ryZ * Math.sin(angle) + p.y * Math.cos(angle)
      const yZ = ryZ * Math.cos(angle) - p.y * Math.cos(angle)

      const scale = 200 / (200 + yZ)
      const screenX = ryX * scale + center
      const screenY = yX * scale + center

      // Calculate opacity based on life and depth (Z)
      const depthAlpha = Math.max(0, (yZ + radius) / (radius * 2)); // Faint on back
      const opacity = p.life * depthAlpha * 0.7;

      if (opacity <= 0.05) continue; // Skip hidden/faded points

      // Draw the point (a faint blue pulse)
      ctx.fillStyle = `rgba(0, 255, 255, ${opacity})`;
      ctx.beginPath();
      // Use the pulseFactor to subtly vary the point size
      ctx.arc(screenX, screenY, 1.5 + (1 - p.life) * p.pulseFactor, 0, Math.PI * 2);
      ctx.fill();

      // Fancy connection: Find *one* nearest neighbor vertex and connect with a faint line
      let minDst = 1000;
      let targetV = null;
      for(let j=0; j<projected2D.length; j++) {
        const v = projected2D[j];
        if (v.z > p.z) continue; // Only connect 'down' in depth
        const dst = Math.pow(v.x-screenX, 2) + Math.pow(v.y-screenY, 2);
        if(dst < minDst && dst < 500) { // arbitrary limit to prevent long crazy lines
          minDst = dst;
          targetV = v;
        }
      }

      if(targetV) {
        ctx.strokeStyle = `rgba(0, 255, 255, ${opacity * 0.9})`;
        ctx.lineWidth = 0.5;
        ctx.beginPath();
        ctx.moveTo(screenX, screenY);
        ctx.lineTo(targetV.x, targetV.y);
        ctx.stroke();
      }
    }

    requestAnimationFrame(draw)
  }

  draw()
})
</script>

  materialSupply.value = Math.random()

  // --- ISSUE DETECTION ---
  if (weatherImpact.value > 0.75) {
    issue.value = "weather"
  }
  else if (materialSupply.value < 0.3) {
    issue.value = "materials"
  }
  else if (workers.value < 8) {
    issue.value = "labor"
  }
  else {
    issue.value = null

  if (issue.value === "materials") speed *= 0.5
  if (issue.value === "labor") speed *= 0.6

  if (

  }
  else {
    logs.value.unshift("Site progressing normally")

<script setup>
import { onMounted } from 'vue'

onMounted(() => {
  const canvas = document.getElementById("sphereCanvas")
  if (!canvas) return

  const ctx = canvas.getContext("2d")
  const size = 350 // Match the SCSS size
  canvas.width = size
  canvas.height = size
  const center = size / 2

  // --- Core Sphere Config ---
  const radius = 100
  const totalRings = 10
  const stepsPerRing = 18

  // --- Dynamic Points (The 'Fancy' part) ---
  const dynamicPoints = []
  const maxDynamicPoints = 15 // Keep it sparse

  // Pre-generate static vertices
  const vertices = []
  for (let i = 0; i <= totalRings; i++) {
    const lat = Math.PI * (i / totalRings - 0.5) // Latitude -90 to 90
    for (let j = 0; j < stepsPerRing; j++) {
      const lon = 2 * Math.PI * (j / stepsPerRing) // Longitude 0 to 360
      const x = radius * Math.cos(lat) * Math.cos(lon)
      const y = radius * Math.cos(lat) * Math.sin(lon)
      const z = radius * Math.sin(lat)
      vertices.push({ x, y, z })
    }
  }

  // Generate real random 3D points *on* the sphere surface
  function generateDynamicPoint() {
    const phi = Math.random() * Math.PI * 2;
    const theta = Math.acos((Math.random() * 2) - 1);
    const pointRadius = radius; // Start on surface

    return {
      x: pointRadius * Math.sin(theta) * Math.cos(phi),
      y: pointRadius * Math.sin(theta) * Math.sin(phi),
      z: pointRadius * Math.cos(theta),
      life: Math.random() * 0.5 + 0.5, // 0.5s to 1s life
      pulseFactor: Math.random() * 2 // How much it 'projects'
    }
  }

  let angle = 0

  function draw() {
    ctx.clearRect(0, 0, canvas.width, canvas.height)
    angle += 0.005 // Slower, purposeful spin

    // --- 1. Project and Rotate all Static Vertices ---
    const projected2D = vertices.map(v => {
      // Basic 3D rotation matrix application
      // Rotate Y (Vertical)
      const ryX = v.x * Math.cos(angle) + v.z * Math.sin(angle)
      const ryZ = -v.x * Math.sin(angle) + v.z * Math.cos(angle)

      // Rotate X (Horizontal)
      const yX = ryZ * Math.sin(angle) + v.y * Math.cos(angle)
      const yZ = ryZ * Math.cos(angle) - v.y * Math.sin(angle)

      // Perspective projection (simple, good enough for wireframe)
      const scale = 200 / (200 + yZ)
      const screenX = ryX * scale + center
      const screenY = yX * scale + center
      return { x: screenX, y: screenY, z: yZ }
    })

    // --- 2. Draw Wireframe Lines ---
    ctx.strokeStyle = "rgba(255, 255, 255, 0.15)"
    ctx.lineWidth = 0.5
    ctx.beginPath()

    // Draw Latitudes
    for (let i = 0; i < totalRings; i++) {
      const start = i * stepsPerRing
      ctx.moveTo(projected2D[start].x, projected2D[start].y)
      for (let j = 1; j < stepsPerRing; j++) {
        ctx.lineTo(projected2D[start + j].x, projected2D[start + j].y)
      }
      ctx.lineTo(projected2D[start].x, projected2D[start].y) // Close ring
    }

    // Draw Longitudes
    for (let j = 0; j < stepsPerRing; j++) {
      ctx.moveTo(projected2D[j].x, projected2D[j].y)
      for (let i = 1; i < totalRings; i++) {
        ctx.lineTo(projected2D[i * stepsPerRing + j].x, projected2D[i * stepsPerRing + j].y)
      }
    }
    ctx.stroke()

    // --- 3. Manage and Draw Dynamic 'Micro-grid' Points ---
    if (dynamicPoints.length < maxDynamicPoints && Math.random() > 0.95) {
      dynamicPoints.push(generateDynamicPoint());
    }

    for (let i = dynamicPoints.length - 1; i >= 0; i--) {
      const p = dynamicPoints[i];
      p.life -= 0.01; // Fade life

      if (p.life <= 0) {
        dynamicPoints.splice(i, 1);
        continue;
      }

      // Project the dynamic point (with the current rotation)
      const ryX = p.x * Math.cos(angle) + p.z * Math.sin(angle)
      const ryZ = -p.x * Math.sin(angle) + p.z * Math.cos(angle)
      const yX = ryZ * Math.sin(angle) + p.y * Math.cos(angle)
      const yZ = ryZ * Math.cos(angle) - p.y * Math.cos(angle)

      const scale = 200 / (200 + yZ)
      const screenX = ryX * scale + center
      const screenY = yX * scale + center

      // Calculate opacity based on life and depth (Z)
      const depthAlpha = Math.max(0, (yZ + radius) / (radius * 2)); // Faint on back
      const opacity = p.life * depthAlpha * 0.7;

      if (opacity <= 0.05) continue; // Skip hidden/faded points

      // Draw the point (a faint blue pulse)
      ctx.fillStyle = `rgba(0, 255, 255, ${opacity})`;
      ctx.beginPath();
      // Use the pulseFactor to subtly vary the point size
      ctx.arc(screenX, screenY, 1.5 + (1 - p.life) * p.pulseFactor, 0, Math.PI * 2);
      ctx.fill();

      // Fancy connection: Find *one* nearest neighbor vertex and connect with a faint line
      let minDst = 1000;
      let targetV = null;
      for(let j=0; j<projected2D.length; j++) {
        const v = projected2D[j];
        if (v.z > p.z) continue; // Only connect 'down' in depth
        const dst = Math.pow(v.x-screenX, 2) + Math.pow(v.y-screenY, 2);
        if(dst < minDst && dst < 500) { // arbitrary limit to prevent long crazy lines
          minDst = dst;
          targetV = v;
        }
      }

      if(targetV) {
        ctx.strokeStyle = `rgba(0, 255, 255, ${opacity * 0.9})`;
        ctx.lineWidth = 0.5;
        ctx.beginPath();
        ctx.moveTo(screenX, screenY);
        ctx.lineTo(targetV.x, targetV.y);
        ctx.stroke();
      }
    }

    requestAnimationFrame(draw)
  }

  draw()
})
</script>

    load.value = Number((1.5 + Math.random() * 2).toFixed(2))

    const rand = Math.random()

    if (rand > 0.92) {
      fault.value = "low_irradiance"
    } else if (rand > 0.85) {
      fault.value = "high_load"
    } else {
      fault.value = null
    }

    // --- SYSTEM RESPONSE ---
    let chargeRate = 0

    if (fault.value === "low_irradiance") {
      irradiance.value *= 0.5
      chargeRate = -0.6
    }
    else if (fault.value === "high_load") {
      load.value *= 1.5
      chargeRate = -0.8
    }
    else {
      chargeRate = output.value > load.value ? 0.6 : -0.4

    // --- LOGS WITH MEANING ---
    if (fault.value === "low_irradiance") {
      logs.value.unshift("Cloud cover detected — generation drop")
    }
    else if (fault.value === "high_load") {
      logs.value.unshift("Load spike detected — battery strain")
    }
    else {
      logs.value.unshift("System stable")
    }

    if (logs.value.length > 5) logs.value.pop()

  }, 2000)
